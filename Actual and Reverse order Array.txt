#include<iostream>
using namespace std;
int main(){
int num[5];
for(int i=0;i<5;i++){
cout<<"Enter an integer: ";
cin>>num[i];
}
cout<<"The array in actual order:\n";
for(int i=0;i<5;i++){
cout<<num[i]<<" ";
}
cout<<"\nThe array in reverse order:\n";
for(int i=4;i>=0;i--){
cout<<num[i]<<" ";
}
return 0;
}
